34 int checkData(T* inVec1, T* inVec2, int numVals)
35 {
36  int checkSum = 0;
37  for (int n = 0; n < numVals; ++n)
38  {
39  checkSum += std::abs(inVec1[n] - inVec2[n]);
40  }
41  if (checkSum == 0)
42  {
43  //std::cout << "Test Passed\n";
44  return 0;
45  }
46  else
47  {
48  //std::cout << "Test Failed\n";
49  return 1;
50  }
51 }
